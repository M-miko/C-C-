#include<stdio.h>
int main()
{ //0000 0000|0000 1111|0100 0010|0011 1111|
	int a=1;//999999      |0|0xf|0x42|0x3f|  
	char *p=&a;//每一次只能移动一个字节；
	*p=0x3f;//用小端存储，低地址存低位；intel（因特尔）//大端，低地址存高位；IBM
	*(p+1)=0x42;//63 66 15 0 /10进制输入也可以
	*(p+2)=0xf;
	*(p+3)=0;
	printf("%d\n",a);//a不能换成*p 因为前面定义了*p=0x3f
return 0;
}
